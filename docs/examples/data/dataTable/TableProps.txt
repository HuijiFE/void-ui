<template>
  <div>
  <vd-data-table :body-data="scoreData" :head-data="HeadData">
  <div class="a" slot="body-row-name" slot-scope="{bodyItem, bodyCell}">自定义 {{bodyCell}}</div>
  <div class="a" slot="body-row-zhihu" slot-scope="{bodyItem, bodyCell}">5555 </div>
</vd-data-table>
  </div>
</template>

<script lang="ts">
import {
  Component,
  Emit,
  Inject,
  Model,
  Prop,
  Provide,
  Vue,
  Watch,
} from 'vue-property-decorator';
export default class VdTableProps extends Vue {
  scoreData = this.getScoreData();

  get HeadData() {
    return Object.keys(this.scoreData[0]).map(k => {
      return {
        // 表头显示的值
        text: k.toUpperCase(),
        // 对应的字段名
        value: k,
      };
    });
  }

  getScoreData() {
    let names = '赵钱孙李周吴郑王';
    let randomScore = () => Math.ceil(Math.random() * 1000000);

    let tableData = names.split('').map(v => {
      return {
        name: `大${v}`,
        zhihu: randomScore(),
        douban: randomScore(),
        wangyi: randomScore(),
      };
    });
    return tableData;
  }
}
</script>
